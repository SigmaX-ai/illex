33 // Need to roll our own distributions, since the STL distributions give implementation
34 // defined results and may differ between platforms.
35 // Also see: https://stackoverflow.com/questions/34903356
36 template <typename T>
37 class UniformIntDistribution {
38  public:
39  typedef T result_type;
40 
41  explicit UniformIntDistribution(T min = std::numeric_limits<T>::min(),
42  T max = std::numeric_limits<T>::max())
43  : min_(min), max_(max) {}
44 
45  void reset() {}
46 
47  template <class G>
48  auto operator()(G& gen) -> T {
49  auto gen_range = static_cast<double>(gen.max() - gen.min());
50  auto scale = (max_ - min_) / (gen_range + 1);
51  return min_ + static_cast<T>((gen() - gen.min()) * scale);
52  }
53 
54  [[nodiscard]] auto min() const -> T { return min_; }
55  [[nodiscard]] auto max() const -> T { return max_; }
56 
57  protected:
58  T min_;
59  T max_;
60 };
