33 // Need to roll our own distributions, since the STL distributions give implementation
34 // defined results and may differ between platforms.
35 // Also see: https://stackoverflow.com/questions/34903356
37 template <typename T>
38 class UniformIntDistribution {
39  public:
40  typedef T result_type;
41 
42  explicit UniformIntDistribution(T min = std::numeric_limits<T>::min(),
43  T max = std::numeric_limits<T>::max())
44  : min_(min), max_(max) {}
45 
46  void reset() {}
47 
48  template <class G>
49  auto operator()(G& gen) -> T {
50  auto gen_range = static_cast<double>(gen.max() - gen.min());
51  auto scale = (max_ - min_) / (gen_range + 1);
52  return min_ + static_cast<T>((gen() - gen.min()) * scale);
53  }
54 
55  [[nodiscard]] auto min() const -> T { return min_; }
56  [[nodiscard]] auto max() const -> T { return max_; }
57 
58  protected:
59  T min_;
60  T max_;
61 };
